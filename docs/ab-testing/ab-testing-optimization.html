Ronald Fisher chose p < 0.05 in 1925 to analyze crop yields. A century later, it’s still the default for product experimentation — not because it’s optimal, but because it was never questioned. Fisher himself noted:

“The calculation is absurdly academic, for in fact no scientific worker has a fixed level of significance at which from year to year, and in all circumstances, he rejects hypotheses.”

— Ronald Fisher, Statistical Methods and Scientific Inference (1956)

What Drives Stopping Criteria?

The stopping rule that minimizes regret isn’t universal — it depends on the economics of your specific test. Four inputs determine it:

The pre-experiment conversion rate. Sets the variance of your test statistic and determines how much room for improvement exists.
Daily users in the experiment and their control/treatment split. More traffic means faster evidence. Uneven splits trade speed for risk protection.
How long the winning variant runs after the decision. A longer deployment horizon means more value at stake — justifying a more aggressive stopping rule.
The realistic lift you expect. Larger effects are detectable faster. Smaller effects require more traffic or more tolerance for uncertainty.

import simulationengine as se

# ── Experiment Assumptions ─────────────────────────────────────────
CONTROL_MEAN       = 0.65      # pre-experiment conversion rate
MAX_TEST_N         = 100_000   # total users in the experiment
SPLIT              = 0.50      # treatment allocation (50/50)
NUMBER_TEST_DAYS   = 20        # maximum experiment duration
POST_TEST_N        = 500_000   # users exposed after decision is made
NUMBER_SIMULATIONS = 10_000    # simulations to run
SEED               = 42

df = se.create_simulation_df(
    number_test_days=NUMBER_TEST_DAYS,
    number_simulations=NUMBER_SIMULATIONS,
    seed=SEED,
    condition_parameters=[
        {
            "mean": CONTROL_MEAN,
            "trials": int(MAX_TEST_N * SPLIT),
        },
    ],
)

import json

import json
import plotly.graph_objects as go
import numpy as np
import polars as pl
from plotly.subplots import make_subplots
from IPython.display import HTML

# ── Time series data (first 100 sims, cumulative conversion rate) ─
N_SIMS = 100
plot_df = (
    df.filter(pl.col("simulation") <= N_SIMS)
    .sort(["simulation", "day"])
    .with_columns([
        pl.col("condition_0_value").cum_sum().over("simulation").alias("cum_value"),
        pl.col("condition_0_trials").cum_sum().over("simulation").alias("cum_trials"),
    ])
    .with_columns(
        (pl.col("cum_value") / pl.col("cum_trials")).alias("conversion_rate")
    )
)

colors = [
    f"rgba(15, 70, 160, {0.4 + 0.3 * np.random.random()})"
    for i in range(N_SIMS)
]

# ── Pre-compute histogram bins for all 10k sims × 20 days ────────
all_df = (
    df.sort(["simulation", "day"])
    .with_columns([
        pl.col("condition_0_value").cum_sum().over("simulation").alias("cum_value"),
        pl.col("condition_0_trials").cum_sum().over("simulation").alias("cum_trials"),
    ])
    .with_columns(
        (pl.col("cum_value") / pl.col("cum_trials")).alias("conversion_rate")
    )
)
hist_data = {}
global_min = float(all_df["conversion_rate"].min())
global_max = float(all_df["conversion_rate"].max())

for day in range(1, NUMBER_TEST_DAYS + 1):
    rates = all_df.filter(all_df["day"] == day)["conversion_rate"].to_numpy()
    counts, edges = np.histogram(rates, bins=50)
    bin_centers = [float((edges[i] + edges[i+1]) / 2) for i in range(len(counts))]
    hist_data[str(day)] = {"x": bin_centers, "y": counts.tolist()}

# ── Build subplot figure ──────────────────────────────────────────
fig = make_subplots(
    cols=2,
    column_widths=[0.58, 0.42],
    subplot_titles=[
        f"Conversion Rate by Day (100 Sims)",
        "Distribution at Day 1 (10k Sims)",
    ],
)

# Left: time series traces (all hidden initially)
for i, sim in enumerate(range(1, N_SIMS + 1)):
    sim_df = plot_df.filter(plot_df["simulation"] == sim)
    fig.add_trace(go.Scatter(
        x=sim_df["day"].to_list(),
        y=sim_df["conversion_rate"].to_list(),
        mode="lines",
        line=dict(color=colors[i], width=2.5),
        opacity=0,
        showlegend=False,
        name=f"Sim {sim}",
        hovertemplate="Sim %{fullData.name} · %{y:.2%}<extra></extra>",
    ), col=1, row=1)

# True mean reference line on left
fig.add_hline(
    y=CONTROL_MEAN,
    line_dash="dash",
    line_color="rgba(20, 60, 160, 0.8)",
    annotation_text=f"True mean ({CONTROL_MEAN:.0%})",
    annotation_position="bottom right",
    col=1, row=1,
)

# Right: histogram for day 1
d1 = hist_data["1"]
fig.add_trace(go.Bar(
    x=d1["x"],
    y=d1["y"],
    marker_color="rgba(15, 70, 160, 0.55)",
    name="Distribution",
    showlegend=False,
    hovertemplate="Rate: %{x:.2%}<br>Count: %{y}<extra></extra>",

fig.add_vline(

    line_color="rgba(20, 60, 160, 0.8)",

(function() {
  function attach() {
    const plots = document.querySelectorAll('.plotly-graph-div');
    const plot  = plots[plots.length - 1];
    if (!plot || !plot._fullData) { setTimeout(attach, 300); return; }

    const simSlider = document.getElementById('sim-slider');
    const simLabel  = document.getElementById('sim-label');
    const daySlider = document.getElementById('day-slider');
    const dayLabel  = document.getElementById('day-label');
    const n         = 100;   // number of sim traces

    // ── Sim slider: show only selected trace ──────────────────────
    function showSim(idx) {
      const upd = { 'line.width': [], 'opacity': [] };
      for (let i = 0; i < n; i++) {
        upd['line.width'].push(i === idx ? 3 : 2);
        upd['opacity'].push(i === idx ? 1.0 : 0.0);
      }
      Plotly.restyle(plot, upd, [...Array(n).keys()]);
    }

    simSlider.addEventListener('input', function() {
      simLabel.textContent = this.value;
      showSim(parseInt(this.value) - 1);
    });

    // ── Day slider: update histogram bars ────────────────────────
    daySlider.addEventListener('input', function() {
      const day = this.value;
      dayLabel.textContent = day;
      const d = window.__histData[day];
      Plotly.restyle(plot, { x: [d.x], y: [d.y] }, [n]);
      Plotly.relayout(plot, { 'annotations[1].text': `Distribution at Day ${day} (10k Sims)` });
    });

    // Pre-select sim 1
    showSim(0);
  }
  attach();
})();
</script>

</div>
